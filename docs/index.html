# 获取代码
# git clone https://github.com/hisey/front-end-doc.git

# 安装 vuepress 
npm install -g vuepress

# 开始写作
vuepress dev 


# 构建静态文件
vuepress build  

# 构建完成之后，将代码推送到 https://github.com/hisey/front-end-doc.git 即可，
# 访问 https://hisey.github.io/front-end-doc/可查看
